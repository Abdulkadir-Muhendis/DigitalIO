161 static inline __attribute__((always_inline))
162 void fastBitWriteSafe(volatile uint8_t* address, uint8_t mask, bool level) {
163  uint8_t s;
164  if (address > (uint8_t*)0X3F) {
165  s = SREG;
166  cli();
167  }
168  if (level) {
169  *address |= mask;
170  } else {
171  *address &= ~mask;
172  }
173  if (address > (uint8_t*)0X3F) {
174  SREG = s;
175  }
176 }
